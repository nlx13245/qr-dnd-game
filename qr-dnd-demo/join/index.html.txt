    <script>
      const params = new URLSearchParams(window.location.search);
      const session = params.get("session") || "UNKNOWN";
      document.getElementById("info").textContent = `Session: ${session}`;

      document.getElementById("joinBtn").addEventListener("click", () => {
        const name = document.getElementById("name").value.trim();
        if (!name) {
          document.getElementById("status").textContent = "Please enter a name.";
          return;
        }

        // Prototype-only: save locally on the player's phone
        localStorage.setItem("playerName", name);
        localStorage.setItem("sessionId", session);

        document.getElementById("status").textContent =
          `âœ… Joined session ${session} as ${name}`;
      });
    </script>
