  <script>
    const params = new URLSearchParams(window.location.search);
    const sessionParamRaw = params.get("session");
    const sessionParam = sessionParamRaw ? sessionParamRaw.toUpperCase() : "";

    const errorEl = document.getElementById("errorMessage");
    const hintEl = document.getElementById("hintText");
    const nameEl = document.getElementById("playerName");
    const joinBtn = document.getElementById("joinBtn");

    function joinGame() {
      const playerName = nameEl.value.trim();

      if (!sessionParam) {
        errorEl.textContent = "‚ùå No session code found. Please scan the QR from the DM screen.";
        hintEl.style.display = "block";
        return;
      }

      if (!playerName) {
        errorEl.textContent = "Please enter your name!";
        return;
      }

      // Save to localStorage (prototype only)
      localStorage.setItem("sessionId", sessionParam);
      localStorage.setItem("playerName", playerName);

      // Show success message
      document.getElementById("displayCode").textContent = sessionParam;
      document.getElementById("displayName").textContent = playerName;
      document.getElementById("formGroup").classList.add("hidden");
      document.getElementById("successMessage").classList.add("show");
    }

    window.addEventListener("load", () => {
      // If opened without session param, show helpful hint (but don't ‚Äúbreak‚Äù the page)
      if (!sessionParam) {
        errorEl.textContent = "‚ÑπÔ∏è You're not in a session yet.";
        hintEl.style.display = "block";
      }
      nameEl.focus();
    });

    joinBtn.addEventListener("click", joinGame);
    nameEl.addEventListener("keydown", (e) => {
      if (e.key === "Enter") joinGame();
    });
  </script>
